const fetchWithProgress = (url, options) => {
  return new Promise((resolve, reject) => {
    options = options || {};
    const xhr = new XMLHttpRequest();
    xhr.open(options.method || 'GET', url);
    for (const key in options.headers || {}) {
      xhr.setRequestHeader(key, (options.headers || {})[key]);
    }
    xhr.onload = (e) => resolve(e.target);
    xhr.onerror = reject;
    if (xhr.upload && options.onProgress) {
      xhr.upload.onprogress = options.onProgress;
    }
    xhr.send(options.body);
  });
};

const response = await fetchWithProgress(url, {

//文件读写权限申请
const Platform = dw.getNativePackage('Platform')
const PermissionsAndroid = dw.getNativePackage('PermissionsAndroid')


//await new Promise((resolve, reject) => setTimeout(resolve, 500))

const checkResultRead = await PermissionsAndroid.check(PermissionsAndroid.PERMISSIONS.READ_EXTERNAL_STORAGE)
const checkResultWrite = await PermissionsAndroid.check(PermissionsAndroid.PERMISSIONS.WRITE_EXTERNAL_STORAGE)

console.log("******打印checkResultRead******", checkResultRead)
console.log("******打印checkResultWrite******", checkResultWrite)

if (Platform.OS === 'android' && ( !checkResultRead || !checkResultWrite)) {
  const modalResult = await dw.showModal({
    title: '权限用途提示',
    content: '是否允许 App 使用文件读写权限以便您使用录像功能',
    showCancel: true,
    cancelText: '取消',
    confirmText: '继续',
    styleType: "wechat"
  })
  if (!modalResult.confirm) {
    return "no"
  }

  const permissions = [PermissionsAndroid.PERMISSIONS.READ_EXTERNAL_STORAGE, PermissionsAndroid.PERMISSIONS.WRITE_EXTERNAL_STORAGE]
  const { status } = await PermissionsAndroid.requestMultiple(permissions)
  console.log("******打印status******", status)
  if (status !== 'granted') {
    dw.showToast({
      title: '请在系统设置允许 App 使用文件读写权限',
    })
    return "no"
  }
}
