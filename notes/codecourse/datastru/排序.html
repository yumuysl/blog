冒泡排序

（一）过程

冒泡排序是一种基于比较的排序算法，它重复地遍历要排序的数列，一次比较两个元素，如果它们的顺序错误就交换位置，直到整个数列都满足了排序的要求。因为排序过程中较大的元素会经过一轮比较慢慢地“冒泡”到数列的最后面，所以这种排序算法被称为冒泡排序。冒泡排序算法的时间复杂度为O(n^2)。过程演示如下： ::: center :::

（二）代码

class Sort{

  //交换位置
  #swap(data,a,b){
    let tmp = data[a];
    data[a] = data[b];
    data[b] = tmp;
    return true;
  }

  //冒泡排序
  maoPaoSort(data){
    for(let i=0; i < data.length; i++){
      for(let j=0; j < data.length-1; j++){
        if(data[j] > data[j+1])  //从小到大排序
          this.#swap(data,j,j+1);
      }
    }
  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//冒泡
paixu.maoPaoSort(myarray)
console.log(myarray);

选择排序

（一）过程

选择排序是一种简单直观的排序算法，其基本思想是每次选择未排序序列中最小（或最大）的元素，放到已排序序列的末尾，时间复杂度为O(n^2)。过程演示如下： ::: center :::

（二）代码

class Sort{

  //交换位置
  #swap(data,a,b){
    let tmp = data[a];
    data[a] = data[b];
    data[b] = tmp;
    return true;
  }

  //选择排序
  selecSort(data){
    let mix;
    for(let i=0; i<data.length-1; i++){
      mix = i;
      for(let j=i; j<data.length; j++){
        if(data[j] < data[mix]){
          mix = j;
        }
      }
      if(data[i] > data[mix]){
        this.#swap(data, mix, i);
      }
    }

  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//选择
paixu.selecSort(myarray);
console.log(myarray);

插入排序

（一）过程

插入排序（Insertion Sort）是一种简单直观的排序算法，其基本思想是将未排序序列中的元素插入到已排序序列中的合适位置。插入排序的时间复杂度为 O(n^2)，空间复杂度为 O(1)。尽管它与选择排序的时间复杂度相同，但插入排序通常比选择排序性能要好，因为插入排序在遇到有序序列时可以提前终止。此外，对于小规模的数据集，插入排序的性能比选择排序更优，因为它的常数因子较小。过程演示如下： ::: center :::

（二）代码

class Sort{

  //插入排序
  insertSort(data){
    let tmp = 0;
    let adr = 0;
    for(let i=1; i < data.length; i++){
      tmp = data[i];
      for(let j=i; j>0; j--){
        //后移
        if(data[j-1]>tmp){
          data[j] = data[j-1];
          adr = j-1;
        }
      }
      if(data[adr]>tmp){
        data[adr] = tmp;
      }
      //console.log(data);
    }
  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//插入
paixu.insertSort(myarray);
console.log(myarray);

归并排序

（一）过程

  1. 归并排序是一种采用分治思想的排序方法，它将待排序的数组分成两个部分，对每个部分分别进行排序，最后将排序好的两个部分合并成一个有序数组。
  2. 通过递归函数完成数组分割和合并，演示如下（图片来源于网络）： ::: center :::

（二）代码

class Sort{

  //归并排序
  //合并
  #merge(left, right){
    //比较
    let i = 0;
    let j = 0;
    const result = [];

    while( i<left.length && j<right.length){
      result.push(
        left[i] <= right[j] ? left[i++] : right[j++]
      );
      //console.log(result)
    }

    //合并为一个数组
    return result.concat( i<left.length ? left.slice(i) : right.slice(j));
  }
  
  //从中间划分
  mergeSort(data){
    //当 data.length <= 1 时会终止递归
    if(data.length > 1) {
      let middle = Math.floor(data.length / 2);
      let left = this.mergeSort(data.slice(0, middle));
      let right = this.mergeSort(data.slice(middle, data.length));
      data = this.#merge(left, right);
    }
    return data;
  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//归并
let result = paixu.mergeSort(myarray);
console.log(result);

快速排序

（一）过程

快速排序是一种分治思想，简单说就是从数据中选任选一共数作为基准，通过筛选将比基准小的数放在左边，比基准大的数放在右边，如此循环，直到只有一个数，最后先从左边连接基准，再连接右边，如此循环连接在一起。演示如下： ::: center :::

（二）代码

class Sort{

  //快速排序
  quirtSort(data){
    if(data.length < 2){
      return data;
    }

    let basies = data[0];
    //左边
    let minData = data.slice(1).filter( item => item < basies );
    //右边
    let maxData = data.slice(1).filter( item => item >= basies );
    return this.quirtSort(minData).concat(basies).concat(this.quirtSort(maxData));
  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//快速
let result = paixu.quirSort(myarray);
console.log(result);

计数排序

（一）过程

计数排序简单理解是通过一个新的数组来完成排列工作，然后再按照这个新数组的排列方式输出，演示如下： ::: center :::

（二）代码

class Sort{
  //找到最大数
  #findMax(data){
    let max = data[0];
    for(let i=0; i<data.length; i++){
      if(max < data[i]){
        max = data[i]
      }
    }
    return max;
  }

  //计数排序
  countSort(data){
    let maxNum = this.#findMax(data);
    let newArrs = new Array(maxNum+1);  //生成具有maxNum+1个元素的空数组
    //console.log(newArrs);
    data.forEach((item,index) => {
      if(!newArrs[item]){
        newArrs[item] = 0;
      }
      newArrs[item]++;
    });

    //合并
    data = [];
    let i=0;
    newArrs.forEach((item,index) => {
      while(item > 0){
        data[i++] = index;
        item--;
      }
    })
    return data;
  }

}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//计数
let result = paixu.countSort(myarray);
console.log(result);

桶排序

（一）过程

桶排序主要是创建n个桶，把相关数字放进相应桶内，各个桶内数字再进行排序，最后按照桶的顺序进行排序，演示如下： ::: center :::

（二）代码

class Sort{
  //计数排序

  //创建桶为桶内数据排序
  #createBucket(data, bucketSize){
    let maxValue = Math.max(...data);
    let minValue = Math.min(...data);

    //桶数
    let bucketNum = Math.floor( (maxValue-minValue) / bucketSize ) + 1; //加 1 表示存储个数的最后一个数

    //把桶准备好,区别各桶
    let buckts = [...Array(bucketNum)].map( () => []);

    //下面效果同上，但更易理解
    // for(let i=0; i<bucketNum; i++){
    //   buckts[i] = [];
    // }

    //把数据分类到对应的桶
    for(let i = 0; i < data.length; i++){
      let index = Math.floor( (data[i] - minValue) / bucketSize );
      buckts[index].push(data[i]);
    }
    return buckts;
  }
  
  //所有数据排序
  #sortMerge(buckets){
    //为每个桶内的数据排序
    for(let i = 0; i < buckets.length; i++){
      this.selecSort(buckets[i]);  //采用上面的选择排序算法
    }

    //拆桶返回数据
    let data = []
    for(let i = 0; i < buckets.length; i++){
      data.push(...buckets[i]);
    }
    return data
  }

  //传入数据，bucketSize表示每个桶里默认有3个数
  bucketSort(data, bucketSize = 3){
    if(data.length < 2){
      return data;
    }

    //创建桶并分装数据 
    const buckets = this.#createBucket(data, bucketSize);

    //排序合并
    return this.#sortMerge(buckets);
  }

}

console.log(myarray);

let paixu = new Sort();

//计数
let result = paixu.countSort(myarray);
console.log(result);

基数排序

（一）过程

基数排序是一种非比较排序算法，其核心思想是先对个位数排序，再对十位数排序，依次类推，演示如下： ::: center :::

（二）代码

class Sort{

  //基数排序
  jishuSort(data){
    const base = 10;
    let divider = 1;  //初始从个位数开始排序
    let maxVal = Math.max(...data);
    while(divider <= maxVal){
      const buckets = [...Array(10)].map(()=>[]);   //创建10个桶
      for(let val of data){
        buckets[ Math.floor( val / divider ) % base ].push(val)
      }
      data = [].concat(...buckets);
      divider *= base;  //位数
    }

    return data;
  }
}

let myarray = [12,45,36,12,11,55,44,2,31,66];
console.log(myarray);

let paixu = new Sort();

//基数
let result = paixu.jishuSort(myarray);
console.log(result);
