一、单链表

（一）什么是单链表，有什么特点

  1. 单链表是由一系列节点组成，每个节点包含一个数据元素和一个指向下一个节点的指针。通常，单链表的第一个节点称为头节点，最后一个节点称为尾节点，尾节点的指针通常为空。
  2. 特点：，每个节点的 next 指针都指向下一个节点，最后一个节点的 next 指针指向 null。

（二）应用场景

适用于频繁插入和删除节点的场景，但是不适用于频繁访问任意节点的场景。

  1. 文件系统：文件的碎片整理，由于文件不是连续存放在内存或存储，操作系统为了把不连续的数据合成一个文件，采用链表将每个文件的位置保存。
  2. 撤销功能：各种文本、图形编辑器中，撤销重做操作家常便饭，这基于单链表良好的删除特性
  3. 实现图、hashMap等一些高级数据结构

（三）封装

//单链表的封装

//定义节点类
class Node {
  constructor(element) {
    this.element = element;
    this.next = null;
  }
}

//定义链表类
class Linelink {
  //定义私有变量
  constructor() {
    this.head = null;  //头节点
    this.count = 0;  //统计链表长度
  }

  //遍历到目标索引
  getindex(index) {
    let current = this.head;
    for (let i = 0; i < index; i++) {
      current = current.next;
    }
    return current;
  }

  //添加到尾部
  addnode(data) {
    let node = new Node(data);

    if (this.head == null) {
      this.head = node;
    } else {
      let current = this.head;
      while (current.next != null) {
        current = current.next;
      }
      current.next = node;
    }
    this.count++;
    return true;
  }

  //插入到指定位置
  insert(place, data) {
    if (place >= 0 && place < this.count) {
      let node = new Node(data)
      let current = this.getindex(place);
      
      if(place == 0){
        node.next = current;
        this.head = node;
      }else{
        let prenode = this.getindex(place-1);
        node.next = current;
        prenode.next = node;
      }
  
      this.count++;
      return true;
    } else {
      return "插入索引超出范围，请用addnode()方法";
    }
  }


  //删除
  remove(index) {
    if (index >= 0 && index < this.count) {
      let current = this.head;

      if (index === 0) {
        this.head = current.next;
      } else {
        let prenode = null;

        for (let i = 0; i < index; i++) {
          prenode = current;
          current = current.next;
        }

        prenode.next = current.next
      }

      this.count--;
      return current;
    } else {
      return "超出范围";
    }
  }


  //更新
  update(index, data) {
    if (index >= 0 && index < this.count) {

      let current = this.getindex(index)
      current.element = data;

      return true;
    } else {
      return "超出范围，无法更改"
    }
  }

  //查找
  getFindData(index) {
    let current = this.getindex(index);
    return current.element;
  }

  //清空
  clear() {
    this.head = null;
    this.count = 0;
    return true;
  }

  //链表长度
  size() {
    return this.count;
  }

  //是否为空
  isEmpty() {
    return this.count === 0;
  }

  //判断任意类型数值是否相等
  isEquip(a, b) {
    //第一种：图个方便，可以a===b
    //第二种：此处如果转化成字符串比较，则出现对象{"d": 1, "m":2}与对象{"m":2, "d":1}不等而出错
    return JSON.stringify(a) === JSON.stringify(b);
    //第三种：如果实在介意，可以用imumutable这个库判断任意类型数据是否相等
  }

  //查找链表是否存在某个值，如果存在返回索引
  indexOf(value) {
    let current = this.head;
    for (let i = 0; i < this.count; i++) {
      if (this.isEquip(current.element, value)) {
        return "存在，索引为：" + i;
      }
      current = current.next;
    }
    return "不存在";
  }

}

二、双向链表

（一）什么是双链表？有什么特点？

  1. 双向链表是链表的一种，它与单链表最大的区别是每个节点不仅包含指向下一个节点的指针，还包含指向上一个节点的指针。
  2. 特点：每个节点有两个指针，分别指向是前驱节点和后继节点

⭐注意：

  • 由于双向链表需要维护 Next 和 Prev 两个指针，因此 占用的存储空间比单向链表略大 ，在存储空间有限的情况下可能不适用。另外，双向链表相对于单向链表的实现也更加复杂， 需要考虑前后驱节点的维护以及插入、删除节点时关联节点指针的修改等问题。

（二）应用场景

  1. 双向遍历：在需要双向遍历链表的场景下，使用双向链表比单向链表更加高效。
  2. 某些算法中需要实现有前驱节点访问的需求：例如反转链表、插入排序、LRU Cache等，这些算法中需要访问当前节点的前驱节点，如果使用单向链表需要从头节点开始遍历，效率较低，而使用双向链表可以直接访问前驱节点，效率更高。
  3. 多端口缓存结构：例如浏览器的历史记录功能，可以使用双向链表，从而可以快速地回到上一个历史记录页面。

（三）封装

//双向链表的封装

//定义节点类
class Node {
  constructor(element) {
    this.element = element;
    this.prov = null;   
    this.next = null;
  }
}

//定义链表类
class Doublelink {
  //定义私有变量
  constructor() {
    this.head = null;  //首部头节点
    this.tail = null;  //尾部头节点
    this.count = 0;  //统计链表长度
  }

  //遍历到目标索引
  getIndexNext(index) {
    let current = this.head;
    for (let i = 0; i < index; i++) {
      current = current.next;
    }
    return current;
  }

  getIndexProv(index) {
    let current = this.tail;
    for (let i = 0; i < index; i++) {
      current = current.prov;
    }
    return current;
  }


  //添加到尾部
  addnode(data) {
    let node = new Node(data);

    if (this.head == null) {
      this.head = node;
      this.tail = node;
    } else {
      this.tail.next = node;  //尾部节点后驱指向新节点
      node.prov = this.tail;  //新节点先驱指向尾部
      this.tail = node;   //更新指向尾部头节点的指针
    }
    this.count++;
    return true;
  }

  //插入到指定位置，插入索引从0开始而不是1
  insert(place, data) {
    if (place >= 0 && place < this.count) {
      let node = new Node(data)
      //为缩短路径，对半分，靠近哪边从哪边开始
      if(place <= (this.count -1 ) / 2){
        let current = this.getIndexNext(place);
        if(place == 0){
          node.next = current;
          current.prov = node;
          this.head = node;
        }else{
          let prenode = this.getIndexNext(place-1);
          node.next = current;
          node.prov = prenode;
          current.prov = node;
          prenode.next = node;
        }
      }else{
        place = this.count - 1 - place
        let current = this.getIndexProv(place);
        if(place == 0){
          node.next = current;
          node.prov = current.prov;
          current.prov = node;
        }else{
          let prenode = this.getIndexProv(place+1);
          node.next = current;
          node.prov = prenode;
          current.prov = node;
          prenode.next = node;
        }
      }
  
      this.count++;
      return true;
    } else {
      return "插入索引超出范围，请用addnode()方法";
    }
  }


  //删除，查找从head开始
  remove(index) {
    if (index >= 0 && index < this.count) {
      let current = this.head;

      //删除也可以和上述插入一样，对半分找到后删除，此处不打算采用此方法
      if(this.count === 1){
        this.head = null;
        this.tail = null;
      }
      else if (index === 0 && this.count != 1) {
        this.head = current.next;
        this.head.prov = null;
      } 
      else if(index === this.count-1 && this.count != 1){
        current = this.tail;
        this.tail = current.prov;
        current.prov = null;
        this.tail.next = null;
      }
      else{
        let prenode = null;
        for (let i = 0; i < index; i++) {
          prenode = current;
          current = current.next;
        }
        prenode.next = current.next;
        current.next.prov = prenode;
      }

      this.count--;
      return current.element;
    } else {
      return "超出范围";
    }
  }


  //更新
  update(index, data) {
    if (index >= 0 && index < this.count) {

      let current = this.getIndexNext(index)
      current.element = data;

      return true;
    } else {
      return "超出范围，无法更改"
    }
  }

  //查找
  getFindData(index) {
    let current = this.getIndexNext(index);
    return current.element;
  }

  //清空
  clear() {
    this.head = null;
    this.tail = null;
    this.count = 0;
    return true;
  }

  //链表长度
  size() {
    return this.count;
  }

  //是否为空
  isEmpty() {
    return this.count === 0;
  }

  //判断任意类型数值是否相等
  isEquip(a, b) {
    //第一种：图个方便，可以a===b
    //第二种：此处如果转化成字符串比较，则出现对象{"d": 1, "m":2}与对象{"m":2, "d":1}不等而出错
    return JSON.stringify(a) === JSON.stringify(b);
    //第三种：如果实在介意，可以用imumutable这个库判断任意类型数据是否相等
  }

  //查找链表是否存在某个值，如果存在返回索引
  indexOf(value) {
    let current = this.head;
    for (let i = 0; i < this.count; i++) {
      if (this.isEquip(current.element, value)) {
        return "存在，索引为：" + i;
      }
      current = current.next;
    }
    return "不存在";
  }

}

for (let i = 0; i < 5; i++) {
  double.addnode(i*10);
}

三、循环链表

（一）什么是循环链表，有什么特点

  1. 在单链表基础上，尾结点的next指针指向第一个结点，这样整个链表变成了环形。
  2. 特点：表中最后一个结点的指针域指向头结点，整个链表形成一个环。

（二）应用场景

  1. 循环队列：循环链表可以作为队列的实现方式。因为队列的数据项在队头和队尾进行删除和插入，而在循环链表中，可以在头部和尾部进行插入、删除操作，使得队列的操作更加高效。
  2. 约瑟夫问题：约瑟夫问题是一个古老的问题，描述的是n个人围成一圈，依次从第k个人开始报数，每报到第m个人出圈，后再继续从下一个人开始报数，直到所有人都出圈。这个问题可以通过循环链表实现。
  3. 循环缓冲区：循环缓冲区在嵌入式系统和音频等领域广泛应用。循环链表可以方便地实现循环缓冲区，使之可以循环读写，避免了内存空间的浪费。
  4. 模拟钟表：钟表是一个圆形的结构，因此可以使用循环链表来模拟钟表的数据结构。
  5. 浏览器历史记录：浏览器中的历史记录就是一个循环链表。每当用户打开一个新的页面时，就会将此页面添加到链表的尾部；当用户点击“返回”按钮时，就会重新回到上一个页面，即链表的前一个节点；当用户点击“前进”按钮时，就会前往下一个页面，即链表的后一个节点。

    this.element = element;
    this.next = null;
  }
}

//定义链表类
class Cirlink {
  //定义私有变量
  constructor() {
    this.head = null;  //头节点

    let node = new Node(data);
    let current = null;
    if (this.head == null) {
      this.head = node;
    } else {
      current = this.getindex(this.count - 1)
      current.next = node;
    }
    node.next = this.head;
    this.count++;
    return true;
