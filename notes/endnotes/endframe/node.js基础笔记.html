一、precess模块

以下是 Node.js process 核心模块的常用方法及案例说明：

一、基础信息获取

  1. process.argv
    获取命令行参数（返回数组）

    // node app.js --name=Aliyun --port=3000
console.log(process.argv); 
// 输出: [node路径, 脚本路径, '--name=Aliyun', '--port=3000']

  2. process.env
    访问环境变量

    // 读取环境变量（常用于区分开发/生产环境）
console.log(process.env.NODE_ENV); 
// 输出 development 或 production

  3. process.cwd()
    获取当前工作目录

    console.log(process.cwd()); 
// 输出如: /Users/yourname/projects

二、进程控制

  1. process.exit([code])
    强制退出进程（默认 code=0）

    if (error) {
  process.exit(1); // 非0表示异常退出
}

  2. process.exitCode
    设置退出码（更安全的替代方案）

    process.exitCode = 1; // 进程结束后自动以 code=1 退出

  3. process.kill(pid[, signal])
    发送信号给进程

    process.kill(process.pid, 'SIGTERM'); // 终止当前进程

三、事件监听

  1. 'exit' 事件
    进程退出时触发（同步操作）

    process.on('exit', (code) => {
  console.log(`进程退出，code: ${code}`);
});

  2. 'uncaughtException' 事件
    捕获未处理的异常

    process.on('uncaughtException', (err) => {
  console.error('捕获到未处理异常:', err);
  process.exit(1);
});

  3. 'SIGINT' 信号
    监听 Ctrl+C 操作

    process.on('SIGINT', () => {
  console.log('收到终止信号，优雅退出');
  server.close();
});

四、实用工具

  1. process.nextTick(callback)
    将回调放入事件循环队列头部

    process.nextTick(() => {
  console.log('这个回调会优先执行');
});

  2. process.memoryUsage()
    获取内存使用情况

    const { rss, heapTotal } = process.memoryUsage();
console.log(`内存占用: ${rss} bytes`);

  3. process.platform
    检测操作系统类型

    if (process.platform === 'win32') {
  console.log('Windows 系统');
}

五、综合案例

// 案例：根据环境变量启动不同配置
const port = process.env.PORT || 3000;
const isProd = process.env.NODE_ENV === 'production';

server.listen(port, () => {
  console.log(`Running in ${isProd ? '生产' : '开发'}模式`);
});

// 优雅退出处理
process.on('SIGTERM', () => {
  console.log('收到终止信号');
  server.close(() => {
    console.log('服务已关闭');
    process.exit(0);
  });
});

核心要点：

  • 优先使用 process.exitCode 替代 process.exit()
  • 通过 uncaughtException 防止进程崩溃
  • process.env 常用于配置管理（配合 .env 文件 + dotenv 库）

粘包问题以及解决方案

  1. 粘包问题：由于数据传输速率以及读取写入均有差异，接收端与发送端设有缓冲区，这些缓冲区用于临时存放待读取/写入的数据，提高主机/服务器性能，这样同时也导致一个问题，当数据到缓冲区后数据并不会被立刻消耗而是等待缓冲区数据足够多后一次性消耗，这样数据会粘在一起，这就是粘包问题。
  2. 解决方法：广泛使用的一个方法是给消息头部设置长度，然后读取相应长度作为区分，具体封包解包代码如下:

封装

class MyTransfromCode{
  constructor(){
    this.packageHeaderLen = 4;
    this.seriaNum = 0;
    this.serialLen = 2;
  }

  /**
  * 编码
  * @Description:
  *  1. 先按照指定的长度来申请一片内存空间作为header
  *  2. 通过writeInt16BE方法整体写入内容
  *  3. 连接header和数据内容
    **/
  encode(data,serialNum){
    const body = Buffer.from(data);
    const headerBuf = Buffer.alloc(this.packageHeaderLen);
    headerBuf.writeInt16BE(serialNum || this.seriaNum);
    headerBuf.writeInt16BE(body.length, this.serialLen);
    if(serialNum == undefined){
      this.serialNum++;
    }

    return Buffer.concat([headerBuf, body])
  }

  /**
  * 解码
  * @Description:
  *  1. 截取header长度
  *  2. 截取内容长度
  *  3. 通过readInt16BE()方法解码，返回数据
  **/
  decode(buffer){
    const headerBuf = buffer.subarray(0, this.packageHeaderLen);
    const bodyBuf = buffer.subarray(this.packageHeaderLen);

    return {
      seriaNum: headerBuf.readInt16BE(),
      bodyLength: headerBuf.readInt16BE(this.serialLen),
      body: bodyBuf.toString()
    }
  }

  /**
  * 获取包大小
  * @Description:
  *  1. 当buffer数据长度小于消息头长度时，说明还未传输完毕
  **/
  getPackageLen(buffer){
    if(buffer.length < this.packageHeaderLen){
      return 0;
    }else{
      return this.packageHeaderLen + buffer.readInt16BE(this.serialLen)
    }
  }
}

module.exports = MyTransfromCode;

九、os模块

核心模块的常用方法及案例说明：

基础系统信息

  1. os.platform()
    获取操作系统类型

    console.log(os.platform()); 
// 输出: 'linux'、'darwin'(macOS)、'win32'(Windows)

  2. os.arch()
    获取 CPU 架构

    console.log(os.arch()); // 'x64'、'arm'、'ia32' 等

  3. os.release()
    获取操作系统版本

    console.log(os.release()); 
// 输出: '10.0.19045' (Windows) 或 '22.4.0' (macOS)

系统资源监控

  1. os.cpus()
    获取 CPU 核心信息

    const cpus = os.cpus();
console.log(`CPU 核心数: ${cpus.length}`);
console.log(cpus[0].model); // 输出: 'Intel(R) Core(TM) i7-9750H'

  2. os.totalmem() / os.freemem()
    获取内存总量与空闲内存

    const totalMB = (os.totalmem() / 1024 ** 2).toFixed(1);
const freeMB = (os.freemem() / 1024 ** 2).toFixed(1);
console.log(`内存: ${freeMB}MB 空闲 / ${totalMB}MB 总量`);

  3. os.loadavg()
    获取系统平均负载（仅 Linux/macOS）

    console.log(os.loadavg()); 
// 输出: [1.21, 0.98, 0.75] (1/5/15分钟负载)

路径与用户信息

  1. os.homedir()
    获取当前用户主目录

    console.log(os.homedir()); 
// 输出: '/Users/username' 或 'C:\Users\username'

  2. os.userInfo([options])
    获取当前用户信息

    const user = os.userInfo();
console.log(user.username); // 当前登录用户名

  3. os.tmpdir()
    获取系统临时目录

    console.log(os.tmpdir()); // 输出临时文件夹路径

网络与实用工具

  1. os.networkInterfaces()
    获取网络接口信息

    const nets = os.networkInterfaces();

综合应用案例

// 案例：生成系统健康报告
const os = require('os');

function systemReport() {
  return `
=== 系统健康报告 ===
平台类型: ${os.platform()} ${os.arch()}
系统版本: ${os.release()}
运行时间: ${(os.uptime()/3600).toFixed(1)} 小时
内存状态: ${(os.freemem()/1024**3).toFixed(1)}GB 可用 / ${(os.totalmem()/1024**3).toFixed(1)}GB 总量
CPU 核心: ${os.cpus().length} 核 (型号: ${os.cpus()[0].model})
当前用户: ${os.userInfo().username}
  `;
}

console.log(systemReport());

核心要点：

  • 使用 os.cpus().length 可优化 多进程应用（如 Cluster 模块）
  • os.totalmem() 常用于监控 内存泄漏
  • os.networkInterfaces() 可辅助 网络调试
  • 优先使用 os.homedir() 替代硬编码路径，保证跨平台兼容性
